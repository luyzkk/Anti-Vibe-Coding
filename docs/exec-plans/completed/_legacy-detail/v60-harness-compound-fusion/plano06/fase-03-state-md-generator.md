### Passo 2: Skeleton do helper

```typescript
// anti-vibe-coding/lib/state-md-generator.ts
// 2026-05-11 (Luiz/dev): D32/M13/CA-45 — gera STATE.md deterministicamente do filesystem
import * as fs from 'fs'
import * as path from 'path'
import * as matter from 'gray-matter' // 2026-05-11 (Luiz/dev): mesmo parser de Plano 05 (G10)
import { resolveV6Paths } from './path-resolver-v6'

type ResourceCounts = {
  compoundActive: number
  compoundArchived: number
  adrs: number
  plansActive: number
  plansCompleted: number
  todoOpen: number
  todoDone: number
  todoSkipped: number
}

type RecentEntry = { path: string; mtimeMs: number }

type PendingEntry = { description: string }

/**
 * Regenera `docs/STATE.md` no projectRoot.
 * Idempotente: chamar 2x produz arquivo identico (alem do timestamp do header).
 * Performance alvo: <500ms em projeto com 100 compound notes.
 *
 * @returns path absoluto do arquivo escrito
 */
export async function regenerateStateMd(projectRoot: string): Promise<string> {
  const paths = resolveV6Paths(projectRoot)
  // 2026-05-11 (Luiz/dev): se projeto nao tem layout v6, NAO criar STATE.md (D32 requer docs/)
  if (!fs.existsSync(paths.docs)) {
    throw new Error(`state-md-generator: ${paths.docs} nao existe — projeto nao eh v6`)
  }

  const counts = countResources(paths, projectRoot)
  const recent = listRecent(paths, projectRoot)
  const pending = listPending(paths, projectRoot)

  const content = renderStateMd({ counts, recent, pending, generatedAt: new Date() })

  const statePath = path.join(paths.docs, 'STATE.md')
  fs.writeFileSync(statePath, content, 'utf-8')
  return statePath
}

function countResources(paths: ReturnType<typeof resolveV6Paths>, projectRoot: string): ResourceCounts {
  // 2026-05-11 (Luiz/dev): contagens dirty-cheap via fs.readdirSync — O(N) por diretorio
  const compoundActive = countMdFiles(paths.compound, { excludeDirs: ['_archived'] })
  const compoundArchived = countMdFiles(path.join(paths.compound, '_archived'))
  const adrs = countMdFiles(paths.designDocs, { filterPrefix: 'ADR-' })
  const plansActive = countMdFiles(paths.execPlansActive)
  const plansCompleted = countMdFiles(paths.execPlansCompleted)
  const todoCounts = countTodoItems(path.join(projectRoot, 'TODO.md'))
  return { compoundActive, compoundArchived, adrs, plansActive, plansCompleted, ...todoCounts }
}

function countMdFiles(
  dir: string,
  opts: { excludeDirs?: string[]; filterPrefix?: string } = {}
): number {
  if (!fs.existsSync(dir)) return 0
  const entries = fs.readdirSync(dir, { withFileTypes: true })
  return entries.filter((e) => {
    if (e.isDirectory()) return false
    if (!e.name.endsWith('.md')) return false
    if (e.name === 'README.md' || e.name === 'index.md') return false
    if (opts.filterPrefix && !e.name.startsWith(opts.filterPrefix)) return false
    return true
  }).length
}

function countTodoItems(todoPath: string): { todoOpen: number; todoDone: number; todoSkipped: number } {
  if (!fs.existsSync(todoPath)) return { todoOpen: 0, todoDone: 0, todoSkipped: 0 }
  const content = fs.readFileSync(todoPath, 'utf-8')
  const lines = content.split('\n')
  let open = 0
  let done = 0
  let skipped = 0
  for (const line of lines) {
    if (/^\s*- \[ \]/.test(line)) open++
    else if (/^\s*- \[x\]/i.test(line)) done++
    else if (/^\s*- \[-\]/.test(line)) skipped++
  }
  return { todoOpen: open, todoDone: done, todoSkipped: skipped }
}

function listRecent(
  paths: ReturnType<typeof resolveV6Paths>,
  projectRoot: string
): RecentEntry[] {
  // 2026-05-11 (Luiz/dev): agrega 4 fontes + TODO.md, ordena por mtime desc, top 5
  // 06-A3: tie-breaker por path alfabetico decrescente (estavel cross-platform)
  const sources = [
    paths.compound,
    paths.designDocs,
    paths.execPlansActive,
    paths.execPlansCompleted,
  ]
  const entries: RecentEntry[] = []
  for (const dir of sources) {
    if (!fs.existsSync(dir)) continue
    for (const e of fs.readdirSync(dir, { withFileTypes: true })) {
      if (!e.isFile() || !e.name.endsWith('.md')) continue
      if (e.name === 'README.md' || e.name === 'index.md') continue
      const full = path.join(dir, e.name)
      const stat = fs.statSync(full)
      entries.push({ path: path.relative(projectRoot, full), mtimeMs: stat.mtimeMs })
    }
  }
  const todoPath = path.join(projectRoot, 'TODO.md')
  if (fs.existsSync(todoPath)) {
    entries.push({
      path: 'TODO.md',
      mtimeMs: fs.statSync(todoPath).mtimeMs,
    })
  }

  // 2026-05-11 (Luiz/dev): sort estavel — mtime desc; tie por path alfabetico desc
  entries.sort((a, b) => {
    if (b.mtimeMs !== a.mtimeMs) return b.mtimeMs - a.mtimeMs
    return b.path.localeCompare(a.path)
  })

  return entries.slice(0, 5)
}

function listPending(
  paths: ReturnType<typeof resolveV6Paths>,
  projectRoot: string
): PendingEntry[] {
  // 2026-05-11 (Luiz/dev): pending = (planos com tag pending-capture) + (todos abertos sem [x] nem [-])
  const out: PendingEntry[] = []

  // Planos com tag pending-capture (Plano 05 fase-06 introduziu)
  if (fs.existsSync(paths.execPlansActive)) {
    for (const e of fs.readdirSync(paths.execPlansActive, { withFileTypes: true })) {
      if (!e.isFile() || !e.name.endsWith('.md')) continue
      const full = path.join(paths.execPlansActive, e.name)
      try {
        const parsed = matter(fs.readFileSync(full, 'utf-8'))
        const tags = parsed.data?.tags
        if (Array.isArray(tags) && tags.includes('pending-capture')) {
          const slug = e.name.replace(/\.md$/, '')
          out.push({ description: `Plan \`${slug}\` — pending compound capture (tag \`pending-capture\`)` })
        }
      } catch (err) {
        // 2026-05-11 (Luiz/dev): G11 — frontmatter malformado nao crasha generator
        console.warn(`state-md-generator: skip ${full} — ${(err as Error).message}`)
      }
    }
  }

  // TODOs abertos
  const todoPath = path.join(projectRoot, 'TODO.md')
  if (fs.existsSync(todoPath)) {
    const lines = fs.readFileSync(todoPath, 'utf-8').split('\n')
    lines.forEach((line, idx) => {
      const m = /^\s*- \[ \]\s+(.+)$/.exec(line)
      if (m) {
        const desc = m[1].length > 80 ? m[1].slice(0, 77) + '...' : m[1]
        out.push({ description: `TODO item ${idx + 1}: ${desc}` })
      }
    })
  }

  return out
}

function renderStateMd(args: {
  counts: ResourceCounts
  recent: RecentEntry[]
  pending: PendingEntry[]
  generatedAt: Date
}): string {
  const { counts, recent, pending, generatedAt } = args
  const lines: string[] = []
  lines.push('<!-- AUTO-GENERATED by lib/state-md-generator.ts — edicoes manuais sao sobrescritas -->')
  lines.push(`<!-- Last regenerated: ${generatedAt.toISOString()} -->`)
  lines.push('')
  lines.push('# Project State')
  lines.push('')
  lines.push('## Resources')
  lines.push('')
  lines.push(`- **Compound notes:** ${counts.compoundActive} (in \`docs/compound/\`)`)
  if (counts.compoundArchived > 0) {
    lines.push(`- **Compound archived:** ${counts.compoundArchived} (in \`docs/compound/_archived/\`)`)
  }
  lines.push(`- **ADRs:** ${counts.adrs} (in \`docs/design-docs/\`)`)
  lines.push(`- **Active plans:** ${counts.plansActive} (in \`docs/exec-plans/active/\`)`)
  lines.push(`- **Completed plans:** ${counts.plansCompleted} (in \`docs/exec-plans/completed/\`)`)
  lines.push(
    `- **TODO items:** ${counts.todoOpen} open / ${counts.todoDone} done / ${counts.todoSkipped} skipped`
  )
  lines.push('')
  lines.push('## Recent Activity')
  lines.push('')
  if (recent.length === 0) {
    lines.push('_no recent activity_')
  } else {
    for (const r of recent) {
      lines.push(`- \`${r.path}\``)
    }
  }
  lines.push('')
  lines.push('## Pending')
  lines.push('')
  if (pending.length === 0) {
    lines.push('_nothing pending_')
  } else {
    for (const p of pending) {
      lines.push(`- ${p.description}`)
    }
  }
  lines.push('')
  return lines.join('\n')
}

// CLI entrypoint (chamavel via `bun run lib/state-md-generator.ts $PWD`)
if (import.meta.path === Bun.main) {
  const root = process.argv[2] ?? process.cwd()
  regenerateStateMd(root)
    .then((p) => console.log(`STATE.md regenerated at ${p}`))
    .catch((e) => {
      console.error(e.message)
      process.exit(1)
    })
}
```

```typescript
// anti-vibe-coding/lib/state-md-generator.test.ts
import { describe, it, expect, beforeEach } from 'bun:test'
import * as fs from 'fs'
import * as path from 'path'
import { regenerateStateMd } from './state-md-generator'

const FIXTURE = path.resolve(__dirname, '..', 'tests', 'fixtures', 'v6-state-fixture')

describe('regenerateStateMd', () => {
  it('writes docs/STATE.md with 3 expected sections', async () => {
    const out = await regenerateStateMd(FIXTURE)
    expect(fs.existsSync(out)).toBe(true)
    const content = fs.readFileSync(out, 'utf-8')
    expect(content).toContain('## Resources')
    expect(content).toContain('## Recent Activity')
    expect(content).toContain('## Pending')
  })

  it('counts compound notes excluding _archived/', async () => {
    await regenerateStateMd(FIXTURE)
    const content = fs.readFileSync(path.join(FIXTURE, 'docs', 'STATE.md'), 'utf-8')
    expect(content).toMatch(/Compound notes:\*\* 2/)
    expect(content).toMatch(/Compound archived:\*\* 1/)
  })

  it('counts ADRs by ADR- prefix', async () => {
    await regenerateStateMd(FIXTURE)
    const content = fs.readFileSync(path.join(FIXTURE, 'docs', 'STATE.md'), 'utf-8')
    expect(content).toMatch(/ADRs:\*\* 2/)
  })

  it('lists plans with pending-capture tag in Pending section', async () => {
    await regenerateStateMd(FIXTURE)
    const content = fs.readFileSync(path.join(FIXTURE, 'docs', 'STATE.md'), 'utf-8')
    expect(content).toContain('2026-05-13-feature-x')
    expect(content).toContain('pending-capture')
  })

  it('is idempotent — second call produces identical content except timestamp', async () => {
    await regenerateStateMd(FIXTURE)
    const first = fs.readFileSync(path.join(FIXTURE, 'docs', 'STATE.md'), 'utf-8')
    await regenerateStateMd(FIXTURE)
    const second = fs.readFileSync(path.join(FIXTURE, 'docs', 'STATE.md'), 'utf-8')
    const stripTs = (s: string) => s.replace(/Last regenerated:.*?-->/, '')
    expect(stripTs(first)).toBe(stripTs(second))
  })

  it('throws when projectRoot is not v6 layout', () => {
    expect(regenerateStateMd('/nonexistent/path')).rejects.toThrow(/nao eh v6/)
  })

  it('counts TODO items by checkbox state', async () => {
    await regenerateStateMd(FIXTURE)
    const content = fs.readFileSync(path.join(FIXTURE, 'docs', 'STATE.md'), 'utf-8')
    expect(content).toMatch(/TODO items:\*\* 1 open \/ 1 done \/ 1 skipped/)
  })
})
```

## Gotchas

- **G5 do plano (PostToolUse no Claude Code):** Generator eh chamavel manualmente via `bun run state:regenerate`. Staleness eh aceitavel — usuario decide.
- **G11 do plano (pending-capture reconhecimento):** Parsa frontmatter de planos em `active/` — frontmatter malformado faz `console.warn` mas nao crasha.
- **06-A3 (mtime tie-breaker):** path alfabetico decrescente. Documentado no helper.
- **06-A8 (`_archived/` em Resources):** Conta separadamente como "Compound archived", nao soma com active.
- **06-A9 (parsing YAML em todos os planos):** Cost ~10ms por arquivo. 100 planos = 1s — acima do alvo. Mitigacao: so parsar arquivos em `active/` (planos completed nao precisam de pending-capture). Active geralmente <10 arquivos.
- **Local — gray-matter dependency:** Se Plano 05 nao introduziu `gray-matter`, esta fase deve adicionar:
  ```bash
  bun add gray-matter
  ```
- **Local — `import.meta.path === Bun.main`:** Bun-specific. Se rodar via Node, substituir por `require.main === module`. Como o plugin obriga Bun (CLAUDE.md), manter Bun-specific.
